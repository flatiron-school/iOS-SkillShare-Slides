Function Declaration

Declare them in the header(.h) files

myObject.h


@interface myObject: NSObject {
  int myInt;
}

- (id) init;
- (void) functionWithArg1:(int)arg1 
                  andArg2:(int)arg2;

@end

myObject.m

#import "myObject.h"

@implementation myObject

- (id) init {

  [super init]
  return self;

}

- (void) functionWithArg1:(int)arg1
                  andArg2:(int)arg2 {

  NSLog(@"%i",arg1+arg2);
}

@end
